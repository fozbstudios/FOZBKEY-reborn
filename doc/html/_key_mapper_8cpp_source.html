47 uint16_t KeyMapper::map_button_to_key(uint8_t button){
48  for (uint8_t i = 0; i < (sizeof(keys)/sizeof(key_mapping)); ++i)
49  {
50  if(keys[i].button == button){
51  return keys[i].key;
52  }
53  }
54  return 0; //No button assignment found
55 }

64  result.deltas[i].delta = 0;

66  {
67  result.deltas[count].delta = button_changes.deltas[i];
68  result.deltas[count].key = this->map_button_to_key(i);

70  }
71  }
