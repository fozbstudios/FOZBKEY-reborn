74 //These functions map a left-to-right, bottom-to-top row/column
75 //number to an actual pin number and then do operations on them
76 const unsigned char row_pin_numbers[] = {0,1,2,3,7};
77 const unsigned char column_pin_numbers[] = {0,1,4,5,6,7};
78 //Weakly pull up a row
79 inline void pull_up_row(unsigned char row){
80  PORTB |= 1 << row_pin_numbers[row]; //DD in, value high
81 }
82 //Strongly pull down the column
83 inline void pull_down_column(unsigned char column){
84  DDRF |= 1 << column_pin_numbers[column]; //DD out, value low
85 }
86 //Get the value of a row pin
87 inline bool measure_row(unsigned char row){
88  return PINB & (1 << row_pin_numbers[row]);
89 }
90 //Turns off the pull-up resistors on all rows
91 const unsigned char row_mask = \
92 (1 << row_pin_numbers[0]) | \
93 (1 << row_pin_numbers[1]) | \
94 (1 << row_pin_numbers[2]) | \
95 (1 << row_pin_numbers[3]) | \
96 (1 << row_pin_numbers[4]);
97 const unsigned char row_off_mask = ~row_mask;
98 inline void reset_rows(){
99  PORTB &= row_off_mask; //None of the row pins will pull up any longer.
100 }
101 //Sets all column pins to "read" mode (data direction in)
102 //so they can't pull down any columns
103 const unsigned char column_mask = \
104 (1 << column_pin_numbers[0]) | \
105 (1 << column_pin_numbers[1]) | \
106 (1 << column_pin_numbers[2]) | \
107 (1 << column_pin_numbers[3]) | \
108 (1 << column_pin_numbers[4]) | \
109 (1 << column_pin_numbers[5]);
110 const unsigned char column_off_mask = ~column_mask;
111 inline void reset_columns(){

115 //Simply iterates through all PHYSICAL keys and sets its corresponding value
116 //in the returned ButtonsState to true if the key is pressed.
117 ButtonsState HardwareController::update(uint8_t led_state){
118  ButtonsState result = {};
119 
120  HardwareController::set_green_led(led_state & 1);//Num lock
121  HardwareController::set_blue_led(led_state & 2);//Caps lock
122 
123  //Pull up the 3 function button pins and check if they are pulled down
124  PORTB |= (1<<4);
125  PORTC |= (1<<7);
126  PORTD |= (1<<6);
127 
128  DDRF &= column_off_mask; //Columns
129  DDRB &= row_off_mask; //Rows
130  PORTF &= column_off_mask;
131  PORTB &= row_off_mask;
132  //Step 1: Pull row up (Weak)
133  //Step 2: Pull column down (Strong)
134  //Step 3: Mesure row pin. If low, key is pressed.
135  //Step 4: Reset all things to in, low.
136 #define row_size 5
137 #define col_size 6
138  unsigned char row;
139  unsigned char col;
140  for(row = 0; row < row_size; row++){
141  pull_up_row(row);
142  for(col = 0; col < col_size; col++){
143  pull_down_column(col);
144  _delay_us(5); //Let signals propogate
145  if(measure_row(row) == 0){ //If the row pin is low, the key is pressed
146  unsigned char button_number = row*col_size + col;
147  result.states[button_number] = true;//Record the button as being pressed
148  }//Scanned specific key
149  reset_columns();
150  }//Scanned all keys in row
151  reset_rows();
152  }//Scanned all keys
153 
154  //The function buttons are not inside the key matrix with the others
155  if(!(PINB & (1<<4))) result.states[63] = true;//button 3
156  if(!(PIND & (1<<6))) result.states[62] = true;//button 2
157  if(!(PINC & (1<<7))) result.states[61] = true;//button 1
158  //Now pull back down function buttons
159  PORTB &= ~(1<<4);
160  PORTC &= ~(1<<7);
161  PORTD &= ~(1<<6);
162  return result;
163 }

165 HardwareController::HardwareController(){
166  //Do NOT use Teensy's LED, as PD6 is currently used for
167  //measuring a key. Pushing the key will short the pin,
168  //which will burn out the pin if it is fully turned
169  //on. The LED pin being weakly pulled up is fine.
170 
171  //Columns (0 to 5, left to right):
172  // PORTF: 0, 1, 4, 5, 6, 7
173  DDRF = 0;
174  PORTF = 0;
175 
176  //Rows (0 to 4, bottom to top):
177  // PORTB: 7, 3, 2, 1, 0
178  DDRB = 0;//Rows
179  PORTB = 0;
180 
181  //Turn on I2C internal pullups. Not really necessary, since we have external pullups too
182  PORTD |= (1<<0) | (1<<1);
183 
184 
185  //Key 1: PB4
186  PORTB |= 1 << 4;//Enable pullup resistor for key 1
187  //Key 2: PD6
188  PORTD |= 1 << 6;//Enable pullup resistor for key 2
189  //Key 3: PC7
190  PORTC |= 1 << 7;//Enable pullup resistor for key 3
191 
192  //LED 1&2 setup
193  TCCR1B = 1 << CS10; //Use the full 16MHz clock as a PWM clock source for timer 1.
194  //For fast PWM, 8 bit, we want to set WGMn2 and WGMn0.
195  TCCR1A = 1 << WGM10;
196  TCCR1B |= 1 << WGM12;
197  //For fast PWM: to clear on compare match with OCRnx and set at TOP, set COMnx1,
198  //where n is timer number and x is channel leter (A/B/C)
199  //Blue led is OC1A (timer 1, channel A)
200  TCCR1A |= 1 << COM1A1;
201  TCCR1A |= 1 << COM1B1;
202  //LED 2 is OCR1A
203  //LED 1 is OCR1B
204 
205  //LED 3 setup
206  //Use full 16MHz clock source for timer 4
207  TCCR4B = 1 << CS40;
208  //To enable fast PWM on timer 4, we have to set PWM4x to 1
209  TCCR4C = 1 << PWM4D;
210  //Set on match, clear on 0

212  //Led 3 is OCR4D
213 }
